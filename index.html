⚡ Cara Kerja Sync 2 Arah
📱➡️📊
HP → Sheets (otomatis)
Setiap transaksi baru, transfer, atau edit akun langsung dikirim ke Google Sheets
📊➡️📱
Sheets → HP (manual / otomatis saat buka)
Klik "Tarik Data" atau app otomatis sync saat pertama dibuka. Cocok untuk 2 HP berbagi data yang sama
⚠️
Penting: 2 HP harus pakai URL yang sama
Copy-paste URL Apps Script yang sama ke kedua HP. Data akan tersinkron lewat Google Sheets sebagai "database" bersama
📋 Panduan Setup Apps Script ▼ Tampilkan
1 Buat Google Sheet Baru

Buka sheets.google.com → buat spreadsheet baru → beri nama "WalletUS". Buat 2 sheet tab: Transactions dan Accounts

2 Pasang Apps Script (versi lengkap dengan GET)

Extensions → Apps Script → hapus semua kode lama → paste ini:

// ═══════════════════════════════════════════════════════
// WalletUS Apps Script v6
// Paste ini ke Google Apps Script, lalu Deploy ulang (New Version)
// ═══════════════════════════════════════════════════════

function doPost(e) {
  try {
    var ss  = SpreadsheetApp.getActiveSpreadsheet();
    // Baca body — works dengan Content-Type: text/plain
    var raw  = (e.postData && e.postData.contents) ? e.postData.contents : '{}';
    var data = JSON.parse(raw);
    var out  = route(ss, data);
    return out;
  } catch(err) {
    return err_(err);
  }
}

function doGet(e) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    // Support ?d=JSON untuk fallback GET
    var raw  = (e.parameter && e.parameter.d) ? decodeURIComponent(e.parameter.d) : '{}';
    var data = JSON.parse(raw);
    if (!data.action) return json_({status:'ok', msg:'WalletUS v6 aktif ✓'});
    return route(ss, data);
  } catch(err) {
    return err_(err);
  }
}

function route(ss, data) {
  if (data.action === 'get_all')            return getData(ss);
  if (data.action === 'push_all')           return pushAll(ss, data);
  if (data.action === 'add_transaction')    return addTx(ss, data);
  if (data.action === 'update_transaction') return updateTx(ss, data);
  if (data.action === 'delete_transaction') return deleteTxAction(ss, data);
  if (data.action === 'sync_accounts')      { writeAccounts(ss, data.accounts); return ok('accounts synced'); }
  if (data.action === 'sync_shopping')      { writeShopping(ss, data.shoppingLists); return ok('shopping synced'); }
  return ok('no action: ' + (data.action || 'undefined'));
}

function getData(ss) {
  return json_({
    status: 'ok',
    transactions:  readSheet(ss, 'Transactions'),
    accounts:      readSheet(ss, 'Accounts'),
    shoppingLists: readShoppingSheet(ss)
  });
}

function pushAll(ss, data) {
  writeTransactions(ss, data.transactions  || []);
  writeAccounts(ss,    data.accounts       || []);
  writeShopping(ss,    data.shoppingLists  || []);
  return ok('push_all done');
}

function addTx(ss, data) {
  var s = getOrCreate(ss, 'Transactions');
  if (s.getLastRow() === 0) {
    s.appendRow(['id','date','desc','category','accountId','accountName','type','amount','note','toAccountId','toAccountName']);
  }
  var t = data.tx || {};
  s.appendRow([t.id||'', t.date||'', t.desc||'', t.category||'', t.accountId||'',
    t.accountName||'', t.type||'', t.amount||0, t.note||'', t.toAccountId||'', t.toAccountName||'']);
  if (data.accounts) writeAccounts(ss, data.accounts);
  return ok('tx added');
}

function updateTx(ss, data) {
  var s    = getOrCreate(ss, 'Transactions');
  var rows = s.getDataRange().getValues();
  var t    = data.tx || {};
  for (var i = 1; i < rows.length; i++) {
    if (String(rows[i][0]) === String(t.id)) {
      s.getRange(i+1, 1, 1, 11).setValues([[
        t.id||'', t.date||'', t.desc||'', t.category||'', t.accountId||'',
        t.accountName||'', t.type||'', t.amount||0, t.note||'', t.toAccountId||'', t.toAccountName||''
      ]]);
      break;
    }
  }
  if (data.accounts) writeAccounts(ss, data.accounts);
  return ok('tx updated');
}

function deleteTxAction(ss, data) {
  var s    = getOrCreate(ss, 'Transactions');
  var rows = s.getDataRange().getValues();
  for (var i = rows.length - 1; i >= 1; i--) {
    if (String(rows[i][0]) === String(data.id)) { s.deleteRow(i + 1); break; }
  }
  if (data.accounts) writeAccounts(ss, data.accounts);
  return ok('tx deleted');
}

function writeTransactions(ss, txs) {
  var s = getOrCreate(ss, 'Transactions');
  s.clearContents();
  s.appendRow(['id','date','desc','category','accountId','accountName','type','amount','note','toAccountId','toAccountName']);
  (txs || []).forEach(function(t) {
    s.appendRow([t.id||'', t.date||'', t.desc||'', t.category||'', t.accountId||'',
      t.accountName||'', t.type||'', t.amount||0, t.note||'', t.toAccountId||'', t.toAccountName||'']);
  });
}

function writeAccounts(ss, accs) {
  var s = getOrCreate(ss, 'Accounts');
  s.clearContents();
  s.appendRow(['id','name','balance','type','color']);
  (accs || []).forEach(function(a) {
    s.appendRow([a.id||'', a.name||'', a.balance||0, a.type||'', a.color||'']);
  });
}

function writeShopping(ss, lists) {
  var s = getOrCreate(ss, 'ShoppingLists');
  s.clearContents();
  s.getRange(1, 1).setValue(JSON.stringify(lists || []));
}

function readSheet(ss, name) {
  var s = getOrCreate(ss, name);
  if (s.getLastRow() <= 1) return [];
  var d   = s.getDataRange().getValues();
  var h   = d[0];
  var out = [];
  for (var i = 1; i < d.length; i++) {
    var o = {};
    h.forEach(function(k, idx) { o[k] = d[i][idx]; });
    if (o.id) out.push(o);
  }
  return out;
}

function readShoppingSheet(ss) {
  var s = getOrCreate(ss, 'ShoppingLists');
  if (s.getLastRow() === 0) return [];
  try {
    var r = s.getRange(1, 1).getValue();
    return r ? JSON.parse(r) : [];
  } catch(e) { return []; }
}

function getOrCreate(ss, name) {
  return ss.getSheetByName(name) || ss.insertSheet(name);
}
function ok(msg)  { return json_({status:'ok',    msg: msg}); }
function err_(e)  { return json_({status:'error', msg: String(e)}); }
function json_(o) {
  return ContentService
    .createTextOutput(JSON.stringify(o))
    .setMimeType(ContentService.MimeType.JSON);
}
3 Deploy Web App

Klik Deploy → New Deployment → Web App
• Execute as: Me
• Who has access: Anyone
Salin URL → paste di kolom di atas → Simpan & Hubungkan

4 HP ke-2

Buka WalletUS di HP ke-2 → halaman Google Sheets → paste URL yang sama → Simpan → klik "Tarik Data dari Sheets". Semua data akan muncul!

🔄 Sinkronisasi Manual

Push: kirim data HP ini ke Sheets (timpa Sheets). Tarik: ambil data dari Sheets ke HP ini (timpa HP).
