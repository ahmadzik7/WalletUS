🔗 Koneksi Apps Script
✓ Setup Apps Script v8 (Drive JSON):
1. Buka script.google.com → proyek yang sudah ada
2. Hapus semua kode lama → paste kode v8 dari panel kanan
3. Deploy → Manage Deployments → Edit ✏ → New Version → Deploy
  • Execute as: Me  |  Who has access: Anyone
4. URL tidak berubah — tidak perlu copy ulang
5. Klik 🔍 Test → harus muncul ✓ v8
6. Klik ⬇ Tarik dari Drive

ℹ Data disimpan sebagai walletUS_data.json di Google Drive kamu.
⚠ Tidak ada batas karakter — semua transaksi aman tersimpan.
📋 Apps Script v7 
// WalletUS Apps Script v8 — Google Drive JSON Storage
// Deploy sebagai Web App: Execute as Me | Who has access: Anyone
// Data disimpan di walletUS_data.json di Google Drive root

var FILE_NAME = 'walletUS_data.json';

function doPost(e) {
  try {
    var raw = '';
    if (e.postData && e.postData.contents) raw = e.postData.contents;
    else if (e.parameter && e.parameter.d) raw = decodeURIComponent(e.parameter.d);
    if (!raw) raw = '{}';
    return route(JSON.parse(raw));
  } catch(err) { return resp({status:'error', msg:String(err)}); }
}

function doGet(e) {
  try {
    var raw = (e.parameter && e.parameter.d) ? decodeURIComponent(e.parameter.d) : '{}';
    var data = JSON.parse(raw);
    if (!data.action) return resp({status:'ok', msg:'WalletUS v8 aktif', version:8});
    return route(data);
  } catch(err) { return resp({status:'error', msg:String(err)}); }
}

function route(data) {
  var a = data.action || '';
  if (a === 'get')    return resp({status:'ok', data: readFile(), version:8});
  if (a === 'save')   { writeFile(data.payload); return resp({status:'ok', msg:'saved'}); }
  if (a === 'ping')   return resp({status:'ok', msg:'WalletUS v8 aktif', version:8});
  return resp({status:'ok', msg:'unknown: '+a});
}

function readFile() {
  var files = DriveApp.getFilesByName(FILE_NAME);
  if (!files.hasNext()) return null;
  var content = files.next().getBlob().getDataAsString();
  return JSON.parse(content);
}

function writeFile(payload) {
  var content = JSON.stringify(payload);
  var files = DriveApp.getFilesByName(FILE_NAME);
  if (files.hasNext()) {
    files.next().setContent(content);
  } else {
    DriveApp.createFile(FILE_NAME, content, MimeType.PLAIN_TEXT);
  }
}

function resp(o) {
  return ContentService
    .createTextOutput(JSON.stringify(o))
    .setMimeType(ContentService.MimeType.JSON);
}
