// WalletUS Apps Script v6
// Deploy sebagai Web App, Execute as: Me, Who has access: Anyone

function doPost(e) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var contents = e.postData && e.postData.contents ? e.postData.contents : '{}';
    return route(ss, JSON.parse(contents));
  } catch(err) { return errResp(err); }
}

function doGet(e) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var d = e.parameter && e.parameter.d ? decodeURIComponent(e.parameter.d) : '{}';
    var data = JSON.parse(d);
    if (!data.action) return jsonResp({status:'ok', msg:'WalletUS v6 aktif'});
    return route(ss, data);
  } catch(err) { return errResp(err); }
}

function route(ss, data) {
  var a = data.action;
  if (a === 'get_all')            return getData(ss);
  if (a === 'push_all')           return pushAll(ss, data);
  if (a === 'add_transaction')    return addTx(ss, data);
  if (a === 'update_transaction') return updateTx(ss, data);
  if (a === 'delete_transaction') return deleteTx(ss, data);
  if (a === 'sync_accounts')      { writeAccounts(ss, data.accounts || []); return okResp('accounts synced'); }
  if (a === 'sync_shopping')      { writeShopping(ss, data.shoppingLists || []); return okResp('shopping synced'); }
  return okResp('unknown action');
}

function getData(ss) {
  return jsonResp({
    status: 'ok',
    transactions: readSheet(ss, 'Transactions'),
    accounts: readSheet(ss, 'Accounts'),
    shoppingLists: readShopping(ss)
  });
}

function pushAll(ss, data) {
  writeTransactions(ss, data.transactions || []);
  writeAccounts(ss, data.accounts || []);
  writeShopping(ss, data.shoppingLists || []);
  return okResp('push_all done');
}

function addTx(ss, data) {
  var s = getOrCreate(ss, 'Transactions');
  var t = data.tx || {};
  if (s.getLastRow() === 0) {
    s.appendRow(['id','date','desc','category','accountId','accountName','type','amount','note','toAccountId','toAccountName']);
  }
  s.appendRow([t.id||'', t.date||'', t.desc||'', t.category||'', t.accountId||'',
    t.accountName||'', t.type||'', t.amount||0, t.note||'', t.toAccountId||'', t.toAccountName||'']);
  if (data.accounts) writeAccounts(ss, data.accounts);
  return okResp('tx added');
}

function updateTx(ss, data) {
  var s = getOrCreate(ss, 'Transactions');
  var rows = s.getDataRange().getValues();
  var t = data.tx || {};
  for (var i = 1; i < rows.length; i++) {
    if (String(rows[i][0]) === String(t.id)) {
      s.getRange(i+1, 1, 1, 11).setValues([[
        t.id||'', t.date||'', t.desc||'', t.category||'', t.accountId||'',
        t.accountName||'', t.type||'', t.amount||0, t.note||'', t.toAccountId||'', t.toAccountName||''
      ]]);
      break;
    }
  }
  if (data.accounts) writeAccounts(ss, data.accounts);
  return okResp('tx updated');
}

function deleteTx(ss, data) {
  var s = getOrCreate(ss, 'Transactions');
  var rows = s.getDataRange().getValues();
  for (var i = rows.length - 1; i >= 1; i--) {
    if (String(rows[i][0]) === String(data.id)) { s.deleteRow(i + 1); break; }
  }
  if (data.accounts) writeAccounts(ss, data.accounts);
  return okResp('tx deleted');
}

function writeTransactions(ss, txs) {
  var s = getOrCreate(ss, 'Transactions');
  s.clearContents();
  s.appendRow(['id','date','desc','category','accountId','accountName','type','amount','note','toAccountId','toAccountName']);
  txs.forEach(function(t) {
    s.appendRow([t.id||'', t.date||'', t.desc||'', t.category||'', t.accountId||'',
      t.accountName||'', t.type||'', t.amount||0, t.note||'', t.toAccountId||'', t.toAccountName||'']);
  });
}

function writeAccounts(ss, accs) {
  var s = getOrCreate(ss, 'Accounts');
  s.clearContents();
  s.appendRow(['id','name','balance','type','color']);
  accs.forEach(function(a) {
    s.appendRow([a.id||'', a.name||'', a.balance||0, a.type||'', a.color||'']);
  });
}

function writeShopping(ss, lists) {
  var s = getOrCreate(ss, 'ShoppingLists');
  s.clearContents();
  s.getRange(1,1).setValue(JSON.stringify(lists || []));
}

function readSheet(ss, name) {
  var s = getOrCreate(ss, name);
  if (s.getLastRow() <= 1) return [];
  var data = s.getDataRange().getValues();
  var headers = data[0];
  var out = [];
  for (var i = 1; i < data.length; i++) {
    var obj = {};
    headers.forEach(function(h, j) { obj[h] = data[i][j]; });
    if (obj.id) out.push(obj);
  }
  return out;
}

function readShopping(ss) {
  var s = getOrCreate(ss, 'ShoppingLists');
  if (s.getLastRow() === 0) return [];
  try {
    var val = s.getRange(1,1).getValue();
    return val ? JSON.parse(val) : [];
  } catch(e) { return []; }
}

function getOrCreate(ss, name) {
  return ss.getSheetByName(name) || ss.insertSheet(name);
}
function okResp(m)  { return jsonResp({status:'ok', msg:m}); }
function errResp(e) { return jsonResp({status:'error', msg:String(e)}); }
function jsonResp(o) {
  return ContentService.createTextOutput(JSON.stringify(o))
    .setMimeType(ContentService.MimeType.JSON);
}
